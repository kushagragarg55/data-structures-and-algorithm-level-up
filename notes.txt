does a linked list cotains a cycle?
public class Solution {
    public ListNode detectCycle(ListNode head) {
        HashMap <ListNode, Boolean> hm = new HashMap <>();
        ListNode temp = head;
        while(temp!=null){
            if(hm.containsKey(temp)) return temp;
            hm.put(temp, true);
            temp = temp.next;
        }
        return null;
    }
}
